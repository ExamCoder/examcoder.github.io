#include <iostream.h>

int main() {
    int num, originalNum, remainder, result = 0, n = 0, i, power, j;
    
    cout << "Enter an integer: ";
    cin >> num;
    
    originalNum = num;

    // Count number of digits
    while (originalNum != 0) {
        originalNum /= 10;
        n++;
    }

    originalNum = num;

    // Calculate sum of digits raised to the power n
    while (originalNum != 0) {
        remainder = originalNum % 10;
        power = 1;
        for (j = 1; j <= n; j++) {
            power *= remainder;  // manual power calculation
        }
        result += power;
        originalNum /= 10;
    }

    if (result == num)
        cout << num << " is an Armstrong number." << endl;
    else
        cout << num << " is not an Armstrong number." << endl;

    return 0;
}
