#include <iostream>
#include <conio.h>
int main() {
    int num, rev = 0, temp;
    cin >> num;
    temp = num;
    while (temp != 0) {
        rev = rev * 10 + temp % 10;
        temp = temp / 10;
    }
    if (num == rev)
        cout << "Palindrome";
    else
        cout << "Not a palindrome";
    getch();
    return 0;
}
